Go SDK

High-performance Go client with goroutine support

Installation

go get github.com/intelcore-ai/go-sdk

Quick Example

package main

import (
    "context"
    "fmt"
    "github.com/intelcore-ai/go-sdk"
)

func main() {
    client := intelcore.NewClient("YOUR_API_KEY")
    
    response, err := client.Chat.Create(context.Background(), &intelcore.ChatRequest{
        Model: "ai-friend-pro",
        Message: "Hello AI!",
    })
    
    if err != nil {
        panic(err)
    }
    
    fmt.Println(response.Content)
}

Java SDK

Enterprise-ready Java library with Spring integration

Installation

<dependency>
  <groupId>com.intelcore</groupId>
  <artifactId>ai-sdk</artifactId>
  <version>1.0.0</version>
</dependency>

Quick Example

import com.intelcore.ai.IntelCoreAI;
import com.intelcore.ai.models.ChatResponse;

IntelCoreAI client = new IntelCoreAI("YOUR_API_KEY");

ChatResponse response = client.chat()
    .model("ai-friend-pro")
    .message("Hello AI!")
    .execute();

System.out.println(response.getContent());
